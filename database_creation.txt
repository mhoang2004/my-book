create table books (
	id int not null auto_increment,
    title text not null,
    image_url text not null,
    book_description text not null,
    book_source text,
    state text not null,
    primary key(id)
);

create table genres (
	book_id int not null,
    genre text not null,
);

create table comments (

	book_id int not null,
    content text not null,
);

create table ratings (
	book_id int not null,
	rating float not null,
);

create table authors (

    author text not null,
	primary key(id)
);

create table writers (

	book_id int not null,
    author_id int not null,
);

SELECT
	books.id,
    books.title,
    books.image_url,
    books.book_description,
    books.book_source,
    books.state,
    authors.id,
    authors.author,
    ratings.rating,
    ratings.votes,
    comments.content AS comment,
    genres.genre
FROM
    books
LEFT JOIN ratings ON books.id = ratings.book_id
LEFT JOIN comments ON books.id = comments.book_id
LEFT JOIN genres ON books.id = genres.book_id
LEFT JOIN writers ON books.id = writers.book_id
LEFT JOIN authors ON writers.author_id = authors.id;
